class Solution {

    int orangesRotting(vector<vector<int>>& grid) {
        vector<int> dx = {-1, 1, 0, 0};

        queue<pair<int,int>> que;
        queue<pair<int,int>> newque; // å®šä¹‰äº†ä¸¤ä¸ªé˜Ÿåˆ—
        int maxrow = grid.size();

        int fresh = 0; // æ–°é²œæ©˜å­
        // å…ˆæŠŠåæ©˜å­å…¥é˜Ÿ
        for(int i=0; i<maxrow; i++){
            for(int j=0;j<maxcol; j++){
                if(grid[i][j] == 2){
                    que.emplace(i,j);
                }else if(grid[i][j] == 1){
                    fresh += 1;
                }
            }
        }
        newque = que;

        // ä¸€è½®ä¸€è½®åœ°æœç´¢è…çƒ‚æ©˜å­å››å‘¨çš„æ–°é²œæ©˜å­

            while(!que.empty()){
                pair<int,int> rot = que.front();que.pop();
                for(int i=0;i<4;i++){
                    int x = rot.first+dx[i];
                    int y = rot.second+dy[i];
                    if(x>=0 && x<maxrow && y>=0 && y<maxcol && grid[x][y]==1){

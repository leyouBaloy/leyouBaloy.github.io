class Solution:
    def lengthOfLongestSubstring(self, s: str) -> int:
        s = s.replace(" ","1")
        size = len(s)
        max = 0
        for i in range(size):
            for j in range(size+1):
                n = len(s[i:j])
                if n==len(set(s[i:j])) and n>max:
                    max = n
        return max
